# LucX MVP Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the LucX MVP — install Xray on servers via SSH, create 2-hop VLESS+Reality chains, generate vless:// client config links. Deployable on desktop, server, and OpenWrt/Keenetic routers.

**Architecture:** Go backend core (chi router, pure-Go SQLite via modernc.org/sqlite, SSH pool, ProxyBackend interface, mode-aware resource management) + Flutter desktop client. Xray configuration via gRPC HandlerService with config.json fallback. Chain engine: pre-flight → execute → commit/rollback. CGO_ENABLED=0 for cross-compilation to MIPS/ARM.

**Tech Stack:** Go 1.23+, chi, modernc.org/sqlite (pure-Go, no CGo), golang-jwt, golang.org/x/crypto, Flutter 3.x, drift, riverpod. Build: CGO_ENABLED=0.

```go
// internal/store/db.go
package store

import (
    "database/sql"
    _ "github.com/mattn/go-sqlite3"
)

type Store struct{ db *sql.DB }

func New(path string) (*Store, error) {
    db, err := sql.Open("sqlite3", path+"?_journal_mode=WAL&_foreign_keys=on")
    if err != nil {
        return nil, err
    }
    db.SetMaxOpenConns(1) // SQLite: single writer
    s := &Store{db: db}
    if err := s.migrate(); err != nil {
        return nil, err
    }
    return s, nil
}

func (s *Store) Close() error { return s.db.Close() }

func (s *Store) migrate() error {
    _, err := s.db.Exec(`
        CREATE TABLE IF NOT EXISTS servers (
            id TEXT PRIMARY KEY,
            name TEXT NOT NULL,
            host TEXT NOT NULL,
            port INTEGER DEFAULT 22,
            username TEXT NOT NULL,
            auth_method TEXT NOT NULL DEFAULT 'password',
            credential TEXT NOT NULL,
            os TEXT DEFAULT '',
            arch TEXT DEFAULT '',
            status TEXT DEFAULT 'unknown',
            source TEXT DEFAULT 'fresh',
            tags TEXT DEFAULT '[]',
            last_seen DATETIME,
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP
        );

        CREATE TABLE IF NOT EXISTS server_backends (
            server_id TEXT REFERENCES servers(id) ON DELETE CASCADE,
            backend_type TEXT NOT NULL,
            version TEXT DEFAULT '',
            status TEXT DEFAULT 'stopped',
            config_path TEXT DEFAULT '',
            api_endpoint TEXT DEFAULT '',
            config_managed BOOLEAN DEFAULT 1,
            installed_at DATETIME,
            PRIMARY KEY (server_id, backend_type)
        );

        CREATE TABLE IF NOT EXISTS chains (
            id TEXT PRIMARY KEY,
            name TEXT NOT NULL,
            status TEXT DEFAULT 'draft',
            applied_at DATETIME,
            created_at DATETIME DEFAULT CURRENT_TIMESTAMP
        );

        CREATE TABLE IF NOT EXISTS chain_nodes (
            chain_id TEXT REFERENCES chains(id) ON DELETE CASCADE,
            server_id TEXT REFERENCES servers(id),
            backend_type TEXT NOT NULL,
            protocol TEXT NOT NULL,
            position INTEGER NOT NULL,
            role TEXT NOT NULL,
            inbound_spec TEXT DEFAULT '{}',
            outbound_spec TEXT DEFAULT '{}',
            inbound_result TEXT DEFAULT '{}',
            outbound_result TEXT DEFAULT '{}',
            PRIMARY KEY (chain_id, position)
        );
    `)
    return err
```

```go
    Status     string    `json:"status"`
    Source     string    `json:"source"`

    CreatedAt  time.Time `json:"created_at"`

type ServerBackend struct {

    BackendType   string `json:"backend_type"`
    Version       string `json:"version"`
    Status        string `json:"status"`
    ConfigPath    string `json:"config_path"`
    APIEndpoint   string `json:"api_endpoint"`

}

    _, err := s.db.Exec(`
        INSERT INTO servers (id, name, host, port, username, auth_method, credential, os, arch, status, source, tags)

    if err != nil {
        return nil, err
    }
    return srv, nil

    rows, err := s.db.Query(`SELECT id, name, host, port, username, auth_method, os, arch, status, source, tags, last_seen, created_at FROM servers ORDER BY created_at DESC`)
    if err != nil {
        return nil, err

    defer rows.Close()

    var servers []Server
    for rows.Next() {
        var srv Server
        if err := rows.Scan(&srv.ID, &srv.Name, &srv.Host, &srv.Port, &srv.Username, &srv.AuthMethod, &srv.OS, &srv.Arch, &srv.Status, &srv.Source, &srv.Tags, &srv.LastSeen, &srv.CreatedAt); err != nil {
            return nil, err
        }
        servers = append(servers, srv)
    }
    return servers, rows.Err()
}
```

```go
// internal/ssh/client.go
package ssh

import (
    "bytes"
    "fmt"
    "net"
    "time"
    "golang.org/x/crypto/ssh"
)

type Client struct {
    conn *ssh.Client
    host string
}

type ConnectParams struct {
    Host       string
    Port       int
    Username   string
    AuthMethod string // "password" or "key"
    Credential string // password or private key PEM
    Timeout    time.Duration
}

func Connect(params ConnectParams) (*Client, error) {
    var authMethods []ssh.AuthMethod
    if params.AuthMethod == "key" {
        signer, err := ssh.ParsePrivateKey([]byte(params.Credential))
        if err != nil {
            return nil, fmt.Errorf("parse key: %w", err)
        }
        authMethods = append(authMethods, ssh.PublicKeys(signer))
    } else {
        authMethods = append(authMethods, ssh.Password(params.Credential))
    }

    config := &ssh.ClientConfig{
        User:            params.Username,
        Auth:            authMethods,
        HostKeyCallback: ssh.InsecureIgnoreHostKey(),
        Timeout:         params.Timeout,
    }

    addr := net.JoinHostPort(params.Host, fmt.Sprintf("%d", params.Port))
    conn, err := ssh.Dial("tcp", addr, config)
    if err != nil {
        return nil, fmt.Errorf("dial %s: %w", addr, err)
    }

    return &Client{conn: conn, host: params.Host}, nil
}

func (c *Client) Exec(cmd string) (string, error) {
    session, err := c.conn.NewSession()
    if err != nil {
        return "", err
    }
    defer session.Close()

    var b bytes.Buffer
    session.Stdout = &b
    if err := session.Run(cmd); err != nil {
        return b.String(), fmt.Errorf("run %q: %w", cmd, err)
    }
    return b.String(), nil
}

func (c *Client) ReadFile(path string) (string, error) {
    return c.Exec("cat " + path)
}

func (c *Client) WriteFile(path, content string) error {
    // Use tee to write as root if needed
    cmd := fmt.Sprintf("tee %s > /dev/null << 'LUCX_EOF'\n%s\nLUCX_EOF", path, content)
    _, err := c.Exec(cmd)
    return err
}

func (c *Client) Host() string { return c.host }
func (c *Client) Close() error { return c.conn.Close() }
```

```go
// internal/ssh/pool.go
package ssh

import (
    "fmt"
    "sync"
    "time"
)

type Pool struct {
    mu      sync.Mutex
    clients map[string]*Client
    params  map[string]ConnectParams
}

func NewPool() *Pool {
    return &Pool{
        clients: make(map[string]*Client),
        params:  make(map[string]ConnectParams),
    }
}

func (p *Pool) Get(serverID string, params ConnectParams) (*Client, error) {
    p.mu.Lock()
    defer p.mu.Unlock()

    if c, ok := p.clients[serverID]; ok {
        // Check connection alive
        if _, err := c.Exec("echo ok"); err == nil {
            return c, nil
        }
        // Dead, close and reconnect
        c.Close()
        delete(p.clients, serverID)
    }

    if params.Timeout == 0 {
        params.Timeout = 10 * time.Second
    }
    client, err := Connect(params)
    if err != nil {
        return nil, fmt.Errorf("connect %s: %w", serverID, err)
    }

    p.clients[serverID] = client
    p.params[serverID] = params
    return client, nil
}

func (p *Pool) Remove(serverID string) {
    p.mu.Lock()
    defer p.mu.Unlock()
    if c, ok := p.clients[serverID]; ok {
        c.Close()
        delete(p.clients, serverID)
    }
}

func (p *Pool) Close() {
    p.mu.Lock()
    defer p.mu.Unlock()
    for id, c := range p.clients {
        c.Close()
        delete(p.clients, id)
    }
}
```

```go
// internal/ssh/distro.go
package ssh

import "strings"

type DistroInfo struct {
    OS       string // "debian", "ubuntu", "almalinux", "arch", "unknown"
    Arch     string // "amd64", "arm64"
    Init     string // "systemd", "openrc"
    PkgMgr   string // "apt", "dnf", "pacman"
}

func DetectDistro(client *Client) (*DistroInfo, error) {
    release, err := client.ReadFile("/etc/os-release")
    if err != nil {
        // Try lsb_release fallback
        release, err = client.Exec("lsb_release -a 2>/dev/null")
        if err != nil {
            return &DistroInfo{OS: "unknown", Arch: "unknown", Init: "unknown", PkgMgr: "unknown"}, nil
        }
    }

    info := &DistroInfo{}
    lower := strings.ToLower(release)

    switch {
    case strings.Contains(lower, "ubuntu"):
        info.OS = "ubuntu"
        info.PkgMgr = "apt"
    case strings.Contains(lower, "debian"):
        info.OS = "debian"
        info.PkgMgr = "apt"
```

```go
// internal/ssh/keys.go
package ssh

import (
    "crypto/aes"
    "crypto/cipher"
    "crypto/rand"
    "crypto/sha256"
    "encoding/base64"
    "errors"
    "io"
    "golang.org/x/crypto/argon2"
)

func EncryptCredential(plaintext, masterPassword string) (string, error) {
    salt := make([]byte, 16)
    if _, err := io.ReadFull(rand.Reader, salt); err != nil {
        return "", err
    }
    key := argon2.IDKey([]byte(masterPassword), salt, 1, 64*1024, 4, 32)

    block, err := aes.NewCipher(key)
    if err != nil {
        return "", err
    }
    aead, err := cipher.NewGCM(block)
    if err != nil {
        return "", err
    }
    nonce := make([]byte, aead.NonceSize())
    if _, err := io.ReadFull(rand.Reader, nonce); err != nil {
        return "", err
    }

    ciphertext := aead.Seal(nil, nonce, []byte(plaintext), nil)
    // Format: base64(salt + nonce + ciphertext)
    combined := append(salt, nonce...)
    combined = append(combined, ciphertext...)
    return base64.StdEncoding.EncodeToString(combined), nil
}

func DecryptCredential(encoded, masterPassword string) (string, error) {
    combined, err := base64.StdEncoding.DecodeString(encoded)
    if err != nil {
        return "", err
    }
    if len(combined) < 16+12+1 {
        return "", errors.New("invalid encrypted data")
    }
    salt := combined[:16]
    nonce := combined[16:28]
    ciphertext := combined[28:]

    key := argon2.IDKey([]byte(masterPassword), salt, 1, 64*1024, 4, 32)
    block, err := aes.NewCipher(key)
    if err != nil {
        return "", err
    }
    aead, err := cipher.NewGCM(block)
    if err != nil {
        return "", err
    }
    plaintext, err := aead.Open(nil, nonce, ciphertext, nil)
    if err != nil {
        return "", errors.New("decryption failed: wrong master password or corrupted data")
    }
    return string(plaintext), nil
}

func DeriveMasterKey(password string) string {
    hash := sha256.Sum256([]byte(password))
    return base64.StdEncoding.EncodeToString(hash[:])
}
```

```go
type VLESSInboundSettings struct {

    Email string `json:"email"`

func (x *XrayBackend) BuildClientConfig(ctx context.Context, ssh backend.SSHClient, inboundTag string) (string, error) {
    // Read the inbound config from the server
    rawCfg, err := x.GetConfig(ctx, ssh)
    if err != nil {
        return "", fmt.Errorf("get config: %w", err)
    }

    // Find the inbound with matching tag
    for _, raw := range rawCfg.Inbounds {
        var inbound struct {
            Tag      string          `json:"tag"`
            Protocol string          `json:"protocol"`
            Port     int             `json:"port"`
            Listen   string          `json:"listen"`
            Settings json.RawMessage `json:"settings"`
            Stream   json.RawMessage `json:"streamSettings"`
        }
        if err := json.Unmarshal(raw, &inbound); err != nil {
            continue
        }
        if inbound.Tag != inboundTag {
            continue
        }

        // Parse settings to get client ID
        var settings VLESSInboundSettings
        if err := json.Unmarshal(inbound.Settings, &settings); err != nil {
            return "", err
        }
        if len(settings.Clients) == 0 {
            return "", fmt.Errorf("no clients in inbound %s", inboundTag)
        }

        client := settings.Clients[0]
        host := ssh.Host()

        // Build vless:// link
        // vless://uuid@host:port?encryption=none&security=reality&type=tcp&...
        u := url.URL{
            Scheme: "vless",
            User:   url.User(client.ID),
            Host:   fmt.Sprintf("%s:%d", host, inbound.Port),
        }

        q := u.Query()
        q.Set("encryption", settings.Decryption)
        if settings.Decryption == "" {
            q.Set("encryption", "none")
        }

        // Parse stream settings
        var stream struct {
            Network  string `json:"network"`
            Security string `json:"security"`
        }
        json.Unmarshal(inbound.Stream, &stream)

        if stream.Security != "" {
            q.Set("security", stream.Security)
        }
        if stream.Network != "" {
            q.Set("type", stream.Network)
        }

        // Add Reality params if present
        var realitySettings struct {
            RealitySettings struct {
                ServerName string `json:"serverName"`
                PublicKey  string `json:"publicKey"`
                ShortID    string `json:"shortId"`
                Fingerprint string `json:"fingerprint"`
            } `json:"realitySettings"`
        }
        json.Unmarshal(inbound.Stream, &realitySettings)
        rs := realitySettings.RealitySettings
        if rs.ServerName != "" {
            q.Set("sni", rs.ServerName)
        }
        if rs.PublicKey != "" {
            q.Set("pbk", rs.PublicKey)
        }
        if rs.ShortID != "" {
            q.Set("sid", rs.ShortID)
        }
        if rs.Fingerprint != "" {
            q.Set("fp", rs.Fingerprint)
        }
        if client.Flow != "" {
            q.Set("flow", client.Flow)
        }

        u.RawQuery = q.Encode()
        return u.String() + "#LucX-" + inboundTag, nil
    }

    return "", fmt.Errorf("inbound %q not found", inboundTag)
```

```go
package store

import "time"

type Chain struct {
    ID        string     `json:"id"`
    Name      string     `json:"name"`
    Status    string     `json:"status"`
    AppliedAt *time.Time `json:"applied_at"`
    CreatedAt time.Time  `json:"created_at"`
    Nodes     []ChainNode `json:"nodes,omitempty"`
}

type ChainNode struct {
    ChainID        string `json:"chain_id"`
    ServerID       string `json:"server_id"`
    BackendType    string `json:"backend_type"`
    Protocol       string `json:"protocol"`
    Position       int    `json:"position"`
    Role           string `json:"role"`
    InboundSpec    string `json:"inbound_spec"`
    OutboundSpec   string `json:"outbound_spec"`
    InboundResult  string `json:"inbound_result"`
    OutboundResult string `json:"outbound_result"`
}

func (s *Store) CreateChain(c *Chain) error {
    tx, err := s.db.Begin()
    if err != nil {
        return err
    }
    defer tx.Rollback()

    _, err = tx.Exec(`INSERT INTO chains (id, name, status) VALUES (?, ?, ?)`,
        c.ID, c.Name, c.Status)
    if err != nil {
        return err
    }
    for _, n := range c.Nodes {
        _, err = tx.Exec(`INSERT INTO chain_nodes (chain_id, server_id, backend_type, protocol, position, role, inbound_spec, outbound_spec, inbound_result, outbound_result) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
            n.ChainID, n.ServerID, n.BackendType, n.Protocol, n.Position, n.Role, n.InboundSpec, n.OutboundSpec, n.InboundResult, n.OutboundResult)
        if err != nil {
            return err
        }
    }
    return tx.Commit()
}

func (s *Store) GetChain(id string) (*Chain, error) {
    c := &Chain{}
    err := s.db.QueryRow(`SELECT id, name, status, applied_at, created_at FROM chains WHERE id = ?`, id).
        Scan(&c.ID, &c.Name, &c.Status, &c.AppliedAt, &c.CreatedAt)
    if err != nil {
        return nil, err
    }

    rows, err := s.db.Query(`SELECT chain_id, server_id, backend_type, protocol, position, role, inbound_spec, outbound_spec, inbound_result, outbound_result FROM chain_nodes WHERE chain_id = ? ORDER BY position`, id)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    for rows.Next() {
        var n ChainNode
        if err := rows.Scan(&n.ChainID, &n.ServerID, &n.BackendType, &n.Protocol, &n.Position, &n.Role, &n.InboundSpec, &n.OutboundSpec, &n.InboundResult, &n.OutboundResult); err != nil {
            return nil, err
        }
        c.Nodes = append(c.Nodes, n)
    }
    return c, rows.Err()
}

func (s *Store) ListChains() ([]Chain, error) {
    rows, err := s.db.Query(`SELECT id, name, status, applied_at, created_at FROM chains ORDER BY created_at DESC`)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    var chains []Chain
    for rows.Next() {
        var c Chain
        if err := rows.Scan(&c.ID, &c.Name, &c.Status, &c.AppliedAt, &c.CreatedAt); err != nil {
            return nil, err
        }
        chains = append(chains, c)
    }
    return chains, rows.Err()
}

func (s *Store) UpdateChainStatus(id, status string) error {
    _, err := s.db.Exec(`UPDATE chains SET status = ?, applied_at = CASE WHEN ? = 'active' THEN CURRENT_TIMESTAMP ELSE applied_at END WHERE id = ?`, status, status, id)
    return err
}

func (s *Store) UpdateChainNodeResult(chainID string, position int, inboundResult, outboundResult string) error {
    _, err := s.db.Exec(`UPDATE chain_nodes SET inbound_result = ?, outbound_result = ? WHERE chain_id = ? AND position = ?`, inboundResult, outboundResult, chainID, position)
    return err
}

func (s *Store) DeleteChain(id string) error {
    _, err := s.db.Exec(`DELETE FROM chains WHERE id = ?`, id)
    return err
```

```go
// internal/chain/planner.go
package chain

import (
    "fmt"
    "github.com/alexeylcp/lucx-core/internal/backend"
    "github.com/alexeylcp/lucx-core/internal/store"
)

type Step struct {
    NodeIndex int              // index in chain.Nodes
    Operation string           // "add_inbound", "add_outbound", "set_routing"
    ServerID  string
    Backend   backend.ProxyBackend
    InboundSpec  *backend.InboundSpec
    OutboundSpec *backend.OutboundSpec
    RoutingRules []backend.RoutingRule
}

type Plan struct {
    Steps []Step
}

func BuildPlan(chain *store.Chain, servers map[string]*store.Server, sshFactory func(serverID string) (backend.SSHClient, error)) (*Plan, error) {
    plan := &Plan{}
    chainID := chain.ID

    for i, node := range chain.Nodes {
        be, err := backend.Get(backend.BackendType(node.BackendType))
        if err != nil {
            return nil, fmt.Errorf("node %d: %w", i, err)
        }

        // Entry node: AddInbound
        if node.Role == "entry" {
            plan.Steps = append(plan.Steps, Step{
                NodeIndex: i,
                Operation: "add_inbound",
                ServerID:  node.ServerID,
                Backend:   be,
                InboundSpec: &backend.InboundSpec{
                    Tag:      fmt.Sprintf("lucx-%s-entry", chainID),
                    Protocol: node.Protocol,
                    Port:     443,
                    Listen:   "0.0.0.0",
                },
            })
        }

        // Hop & Entry (non-exit): AddOutbound to next node + SetRouting
        if node.Role != "exit" && i+1 < len(chain.Nodes) {
            nextNode := chain.Nodes[i+1]
            outboundTag := fmt.Sprintf("lucx-%s-hop%d-to-%d", chainID, i, i+1)
            plan.Steps = append(plan.Steps, Step{
                NodeIndex: i,
                Operation: "add_outbound",
                ServerID:  node.ServerID,
                Backend:   be,
                OutboundSpec: &backend.OutboundSpec{
                    Tag:      outboundTag,
                    Protocol: node.Protocol,
                    // Settings filled in at execution time with next server's address
                },
            })
            plan.Steps = append(plan.Steps, Step{
                NodeIndex: i,
                Operation: "set_routing",
                ServerID:  node.ServerID,
                Backend:   be,
                RoutingRules: []backend.RoutingRule{{
                    Type:        "field",
                    InboundTag:  []string{fmt.Sprintf("lucx-%s-entry", chainID)},
                    OutboundTag: outboundTag,
                }},
            })
        }

        // Hop node: AddInbound (to receive from previous)
        if node.Role == "hop" {
            plan.Steps = append(plan.Steps, Step{
                NodeIndex: i,
                Operation: "add_inbound",
                ServerID:  node.ServerID,
                Backend:   be,
                InboundSpec: &backend.InboundSpec{
                    Tag:      fmt.Sprintf("lucx-%s-hop%d", chainID, i),
                    Protocol: node.Protocol,
                    Port:     443,
                    Listen:   "0.0.0.0",
                },
            })
        }

        // Exit node: AddInbound (final)
        if node.Role == "exit" && i != 0 { // also entry if single-node
            plan.Steps = append(plan.Steps, Step{
                NodeIndex: i,
                Operation: "add_inbound",
                ServerID:  node.ServerID,
                Backend:   be,
                InboundSpec: &backend.InboundSpec{
                    Tag:      fmt.Sprintf("lucx-%s-exit", chainID),
                    Protocol: node.Protocol,
                    Port:     443,
                    Listen:   "0.0.0.0",
                },
            })
        }
    }

    return plan, nil
}
```

```go
// internal/chain/validator.go
package chain

import (
    "context"
    "fmt"
    "github.com/alexeylcp/lucx-core/internal/backend"
)

func Validate(ctx context.Context, plan *Plan, sshFactory func(serverID string) (backend.SSHClient, error)) error {
    // Check all servers are reachable via SSH
    serverSet := make(map[string]bool)
    for _, step := range plan.Steps {
        serverSet[step.ServerID] = true
    }

    for serverID := range serverSet {
        ssh, err := sshFactory(serverID)
        if err != nil {
            return fmt.Errorf("server %s SSH: %w", serverID, err)
        }
        if _, err := ssh.Exec("echo ok"); err != nil {
            return fmt.Errorf("server %s unreachable: %w", serverID, err)
        }
    }

    // Check backend status on each server
    for serverID := range serverSet {
        for _, step := range plan.Steps {
            if step.ServerID != serverID {
                continue
            }
            ssh, _ := sshFactory(serverID)
            status, err := step.Backend.Status(ctx, ssh)
            if err != nil {
                return fmt.Errorf("server %s backend status: %w", serverID, err)
            }
            if !status.Running {
                return fmt.Errorf("server %s: backend %s not running", serverID, step.Backend.Type())
            }
        }
    }

    return nil
}
```

```go
// internal/chain/executor.go
package chain

import (
    "context"
    "encoding/json"
    "fmt"
    "github.com/alexeylcp/lucx-core/internal/backend"
)

type ExecutedStep struct {
    Step      Step
    Inbound   *backend.InboundResult
    Outbound  *backend.OutboundResult
}

func Execute(ctx context.Context, plan *Plan, sshFactory func(serverID string) (backend.SSHClient, error)) ([]ExecutedStep, error) {
    var executed []ExecutedStep

    for _, step := range plan.Steps {
        ssh, err := sshFactory(step.ServerID)
        if err != nil {
            rollback(ctx, executed, sshFactory)
            return nil, fmt.Errorf("step %s: ssh %s: %w", step.Operation, step.ServerID, err)
        }

        es := ExecutedStep{Step: step}

        switch step.Operation {
        case "add_inbound":
            result, err := step.Backend.AddInbound(ctx, ssh, *step.InboundSpec)
            if err != nil {
                rollback(ctx, executed, sshFactory)
                return nil, fmt.Errorf("add_inbound on %s: %w", step.ServerID, err)
            }
            es.Inbound = &result

        case "add_outbound":
            result, err := step.Backend.AddOutbound(ctx, ssh, *step.OutboundSpec)
            if err != nil {
                rollback(ctx, executed, sshFactory)
                return nil, fmt.Errorf("add_outbound on %s: %w", step.ServerID, err)
            }
            es.Outbound = &result

        case "set_routing":
            if err := step.Backend.SetRouting(ctx, ssh, step.RoutingRules); err != nil {
                rollback(ctx, executed, sshFactory)
                return nil, fmt.Errorf("set_routing on %s: %w", step.ServerID, err)
            }
        }

        executed = append(executed, es)
    }

    return executed, nil
}
```

```go
// internal/chain/rollback.go
package chain

import (
    "context"
    "log"
    "github.com/alexeylcp/lucx-core/internal/backend"
)

func rollback(ctx context.Context, executed []ExecutedStep, sshFactory func(serverID string) (backend.SSHClient, error)) {
    log.Printf("ROLLBACK: undoing %d steps", len(executed))
    // Undo in reverse order
    for i := len(executed) - 1; i >= 0; i-- {
        es := executed[i]
        ssh, err := sshFactory(es.Step.ServerID)
        if err != nil {
            log.Printf("ROLLBACK ERROR: cannot connect to %s: %v", es.Step.ServerID, err)
            continue
        }

        switch es.Step.Operation {
        case "add_inbound":
            if es.Inbound != nil {
                if err := es.Step.Backend.RemoveInbound(ctx, ssh, es.Inbound.Tag); err != nil {
                    log.Printf("ROLLBACK ERROR: remove inbound %s: %v", es.Inbound.Tag, err)
                }
            }
        case "add_outbound":
            if es.Outbound != nil {
                if err := es.Step.Backend.RemoveOutbound(ctx, ssh, es.Outbound.Tag); err != nil {
                    log.Printf("ROLLBACK ERROR: remove outbound %s: %v", es.Outbound.Tag, err)
                }
            }
        // set_routing is undone implicitly when inbounds/outbounds are removed
        }
    }
}
```

```go
// internal/chain/engine.go
package chain

import (
    "context"
    "encoding/json"
    "fmt"
    "github.com/alexeylcp/lucx-core/internal/backend"
    "github.com/alexeylcp/lucx-core/internal/store"
)

type Engine struct {
    store      *store.Store
    sshFactory func(serverID string) (backend.SSHClient, error)
}

func NewEngine(s *store.Store, sshFactory func(serverID string) (backend.SSHClient, error)) *Engine {
    return &Engine{store: s, sshFactory: sshFactory}
}

func (e *Engine) Apply(ctx context.Context, chain *store.Chain) error {
    // 1. Build execution plan
    servers := make(map[string]*store.Server) // populated from store
    plan, err := BuildPlan(chain, servers, e.sshFactory)
    if err != nil {
        return fmt.Errorf("plan: %w", err)
    }

    // 2. Pre-flight validation
    if err := Validate(ctx, plan, e.sshFactory); err != nil {
        return fmt.Errorf("validate: %w", err)
    }

    // 3. Execute
    executed, err := Execute(ctx, plan, e.sshFactory)
    if err != nil {
        // Save failed status
        e.store.UpdateChainStatus(chain.ID, "draft")
        return fmt.Errorf("execute: %w", err)
    }

    // 4. Save results to chain_nodes
    for _, es := range executed {
        var inboundJSON, outboundJSON string
        if es.Inbound != nil {
            b, _ := json.Marshal(es.Inbound)
            inboundJSON = string(b)
        }
        if es.Outbound != nil {
            b, _ := json.Marshal(es.Outbound)
            outboundJSON = string(b)
        }
        e.store.UpdateChainNodeResult(chain.ID, es.Step.NodeIndex, inboundJSON, outboundJSON)
    }

    // 5. Commit
    if err := e.store.UpdateChainStatus(chain.ID, "active"); err != nil {
        return fmt.Errorf("commit: %w", err)
    }

    return nil
}
```

```go
// internal/api/auth.go
package api

import (
    "encoding/json"
    "net/http"
    "time"
    "github.com/golang-jwt/jwt/v5"
)

var jwtSecret []byte

func handleLogin(secret string) http.HandlerFunc {
    jwtSecret = []byte(secret)
    return func(w http.ResponseWriter, r *http.Request) {
        var req struct {
            Password string `json:"password"`
        }
        if err := json.NewDecoder(r.Body).Decode(&req); err != nil {
            http.Error(w, `{"error":"invalid json"}`, 400)
            return
        }
        // Simple single-password auth for personal use
        if req.Password != secret {
            http.Error(w, `{"error":"unauthorized"}`, 401)
            return
        }
        token := jwt.NewWithClaims(jwt.SigningMethodHS256, jwt.MapClaims{
            "iat": time.Now().Unix(),
            "exp": time.Now().Add(24 * time.Hour).Unix(),
        })
        tokenStr, _ := token.SignedString(jwtSecret)
        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(map[string]string{"token": tokenStr})
    }
}

func authMiddleware(secret string) func(http.Handler) http.Handler {
    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            tokenStr := r.Header.Get("Authorization")
            if len(tokenStr) < 8 || tokenStr[:7] != "Bearer " {
                http.Error(w, `{"error":"missing token"}`, 401)
                return
            }
            token, err := jwt.Parse(tokenStr[7:], func(t *jwt.Token) (interface{}, error) {
                return []byte(secret), nil
            })
            if err != nil || !token.Valid {
                http.Error(w, `{"error":"invalid token"}`, 401)
                return
            }
            next.ServeHTTP(w, r)
        })
    }
}
```

```dart
// frontend/lib/screens/chains/chain_builder_screen.dart
import 'package:flutter/material.dart';
import 'package:flutter_riverpod/flutter_riverpod.dart';
import '../../models/server.dart';
import '../../models/chain.dart';
import '../../providers/servers_provider.dart';
import '../../providers/chains_provider.dart';

class ChainBuilderScreen extends ConsumerStatefulWidget {
  const ChainBuilderScreen({super.key});

  @override
  ConsumerState<ChainBuilderScreen> createState() => _ChainBuilderScreenState();
}

class _ChainBuilderScreenState extends ConsumerState<ChainBuilderScreen> {
  final _nameCtrl = TextEditingController();
  final List<String> _selectedServerIds = [];

  @override
  Widget build(BuildContext context) {
    final serversAsync = ref.watch(serversProvider);
    return Scaffold(
      appBar: AppBar(title: Text('Chain Builder'), actions: [
        TextButton(
          onPressed: _selectedServerIds.length >= 2 ? _applyChain : null,
          child: Text('Apply', style: TextStyle(color: _selectedServerIds.length >= 2 ? Color(0xFF2EA043) : Colors.grey)),
        ),
      ]),
      body: serversAsync.when(
        data: (servers) => Column(children: [
          Padding(padding: EdgeInsets.all(16), child: TextField(
            controller: _nameCtrl,
            decoration: InputDecoration(labelText: 'Chain Name', hintText: 'e.g. FI → NL → DE'),
          )),
          Padding(padding: EdgeInsets.symmetric(horizontal: 16), child: Text(
            'Select servers in order: Entry → Hop(s) → Exit',
            style: TextStyle(color: Colors.grey, fontSize: 13),
          )),
          SizedBox(height: 12),
          Expanded(child: _buildCanvas(servers)),
          SizedBox(height: 80),
        ]),
        loading: () => Center(child: CircularProgressIndicator()),
        error: (e, _) => Center(child: Text('Error: $e')),
      ),
    );
  }

  Widget _buildCanvas(List<Server> servers) {
    final onlineServers = servers.where((s) => s.isOnline || true).toList(); // allow offline for draft
    return ListView.builder(
      padding: EdgeInsets.symmetric(horizontal: 16),
      itemCount: onlineServers.length,
      itemBuilder: (_, i) {
        final srv = onlineServers[i];
        final isSelected = _selectedServerIds.contains(srv.id);
        final idx = _selectedServerIds.indexOf(srv.id);
        return Card(
          color: isSelected ? Color(0xFF1A3A1A) : null,
          child: ListTile(
            leading: isSelected
                ? CircleAvatar(
                    backgroundColor: idx == 0 ? Color(0xFF4FC3F7) : idx == _selectedServerIds.length - 1 ? Color(0xFF2EA043) : Color(0xFFBC8C4C),
                    child: Text('${idx + 1}', style: TextStyle(fontWeight: FontWeight.bold)),
                  )
                : null,
            title: Text(srv.name),
            subtitle: Text('${srv.host} · ${srv.status}'),
            trailing: isSelected ? Icon(Icons.check_circle, color: Color(0xFF2EA043)) : null,
            onTap: () => setState(() {
              if (isSelected) {
                _selectedServerIds.remove(srv.id);
              } else {
                _selectedServerIds.add(srv.id);
              }
            }),
          ),
        );
      },
    );
  }

  Future<void> _applyChain() async {
    if (_selectedServerIds.length < 2) {
      ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text('Select at least 2 servers')));
      return;
    }
    final nodes = _selectedServerIds.asMap().entries.map((e) => {
      'server_id': e.value,
      'backend_type': 'xray',
      'protocol': 'vless',
      'position': e.key,
      'role': e.key == 0 ? 'entry' : e.key == _selectedServerIds.length - 1 ? 'exit' : 'hop',
    }).toList();

    final data = {'name': _nameCtrl.text.isNotEmpty ? _nameCtrl.text : 'Chain ${DateTime.now().millisecondsSinceEpoch}', 'nodes': nodes};

    try {
      final chain = await ref.read(chainsProvider.notifier).createChain(data);
      await ref.read(chainsProvider.notifier).applyChain(chain.id);
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text('Chain applied!')));
        Navigator.pop(context);
      }
    } catch (e) {
      if (mounted) {
        ScaffoldMessenger.of(context).showSnackBar(SnackBar(content: Text('Error: $e')));
      }
    }
  }
}
```
